 * Events by IDs.
 */
public class EventsByIdsQueryResponse extends GraphQLResult<Map<String, java.util.List<EventTO>>> {

    private static final String OPERATION_NAME = "eventsByIds";

    public EventsByIdsQueryResponse() {
    }

    /**
     * Events by IDs.
     */

    }

}
